for filename in $1/*; do
    if [ -d "$filename" ] ; then
        $0 $filename;
    else
        if [[ $filename == *.xml ]] ; then
          if grep -q res-auto "$filename"; then
            if grep -q "Modified by Vít Skalický 2018" "$filename"; then
              echo "OK          $filename";
            else
              echo $filename
            fi
          fi
        fi
    fi
done
